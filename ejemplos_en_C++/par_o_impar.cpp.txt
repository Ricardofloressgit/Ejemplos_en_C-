#include <iostream>
using namespace std;

int main(int argc, char *argv[]) {
	
	int numero = 0;
	
	// Solicitar al usuario un número
	cout << "Ingresa un número: ";
	cin >> numero;
	
	// Verificar si el número es par o impar
	if (numero % 2 == 0) {
		cout << "el numero "<<numero<<" "<<"es par"<<endl;
	} else {
		cout <<"el numero "<<numero<<" "<<"es impar"<<endl;
	}
	
	return 0;
	return 0;
}
